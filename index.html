    <script>

        let currentStep = 0;

        const totalSteps = 6;


        function init() {

            lucide.createIcons();

            updateUI();

            renderDots();

        }


        function renderDots() {

            const container = document.getElementById('progress-dots');

            const mobileContainer = document.getElementById('mobile-dots');

            container.innerHTML = '';

            mobileContainer.innerHTML = '';


            for (let i = 0; i < totalSteps; i++) {

                const dot = document.createElement('div');

                dot.className = `h-1.5 rounded-full transition-all duration-300 ${i === currentStep ? 'w-8 bg-indigo-600' : 'w-2 bg-slate-200'}`;

                container.appendChild(dot.cloneNode(true));

                mobileContainer.appendChild(dot);

            }

        }


        function updateUI() {

            // Update Navigation Buttons

            const navButtons = document.querySelectorAll('.nav-button');

            navButtons.forEach((btn, index) => {

                if (index === currentStep) {

                    btn.classList.add('active');

                } else {

                    btn.classList.remove('active');

                }

            });


            // Update Content Blocks

            const contents = document.querySelectorAll('.step-content');

            contents.forEach((content, index) => {

                if (index === currentStep) {

                    content.classList.add('active');

                } else {

                    content.classList.remove('active');

                }

            });


            // Update Pagination Controls

            const prevBtn = document.getElementById('prev-btn');

            const nextBtn = document.getElementById('next-btn');

            

            if (currentStep === 0) {

                prevBtn.classList.add('text-slate-300', 'pointer-events-none');

                prevBtn.classList.remove('text-slate-500', 'hover:text-indigo-600');

            } else {

                prevBtn.classList.remove('text-slate-300', 'pointer-events-none');

                prevBtn.classList.add('text-slate-500', 'hover:text-indigo-600');

            }


            if (currentStep === totalSteps - 1) {

                nextBtn.querySelector('span').innerText = "Start Over";

            } else {

                nextBtn.querySelector('span').innerText = "Next Step";

            }


            // Mobile Counter

            document.getElementById('mobile-counter').innerText = `0${currentStep + 1}/06`;


            renderDots();

            window.scrollTo({ top: 200, behavior: 'smooth' });

        }


        function switchStep(index) {

            currentStep = index;

            updateUI();

        }


        function nextStep() {

            if (currentStep < totalSteps - 1) {

                currentStep++;

            } else {

                currentStep = 0;

            }

            updateUI();

        }


        function prevStep() {

            if (currentStep > 0) {

                currentStep--;

                updateUI();

            }

        }


        window.onload = init;

    </script>
